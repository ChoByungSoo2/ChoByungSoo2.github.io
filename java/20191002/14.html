package org.javaro.lecture; public class Pr { public static void main(String[] args) { System.out.println("조병수 "+"20191002"+"-"+"반복문 숙제\n"); System.out.println("while문을 이용,1부터 100까지 합"); int i=1, sum=0; while(i<=100) { sum += i++; } System.out.println("1부터 100까지 합 = "+ sum); System.out.println("\nfor문을 이용,1부터 100까지 합"); sum=0; for(i = 1; i<=100; i++) { sum += i; } System.out.println("1부터 100까지 합 = "+ sum); System.out.println("\nfor문을 이용,삼각형을 출력"); final int N =30; for(i=N; i>0; i--) { for(int j=i; j>0; j--) { System.out.print("*"); } System.out.println(""); } System.out.println("\nfor문을 이용,원을 출력"); final int NN=15; for(i=-NN;i<=NN;i++) { for(int j=-NN;j<=NN;j++) { if(i*i+j*j<=NN*NN) System.out.print("* "); else System.out.print(" "); } System.out.println(""); } } }
