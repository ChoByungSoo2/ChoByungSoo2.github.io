package org.javaro.lecture; public class P7_1 { public static void main(String[] args) { // TODO Auto-generated method stub System.out.println("조병수-20191016-객체"); System.out.println("\n곱하기, 4각형 면적, 3각형 면적 문제"); System.out.println("\n[절차 프로그램]"); double x, y, area; x=10; y=20; System.out.println("곱하기="+x*y); area = x*y; System.out.println("4각형 면적 = "+area); area = 0.5*x*y; System.out.println("3각형 면적 = "+area); System.out.println("\n[함수 프로그램]"); System.out.println("곱하기="+mul(x,y)); area = fn_area(1,x,y); System.out.println("4각형 면적 = "+area); area = fn_area(2,x,y); System.out.println("3각형 면적 = "+area); System.out.println("\n[객체 프로그램]"); P7_2 pr1 = new P7_2(); pr1.b = x; pr1.c = y; System.out.println("곱하기="+pr1.mul()); pr1.a = 1; System.out.println("4각형 면적 = "+pr1.fn_area()); pr1.a = 2; System.out.println("3각형 면적 = "+pr1.fn_area()); System.out.println("\n[자바빈 프로그램]"); P7_2 pr2 = new P7_2(); pr2.setB(x); pr2.setC(y); System.out.println("곱하기="+pr2.mul()); pr2.setA(1); System.out.println("4각형 면적 = "+pr2.fn_area()); pr2.setA(2); System.out.println("3각형 면적 = "+pr2.fn_area()); } static double mul(double a, double b) { return a*b; } static double fn_area(int a, double b, double c) { double result = 0; if(a==1) { result = b*c; }else if(a ==2 ) { result = 0.5*b*c; } return result; } } class P7_2{ int a ; double b, c; void setA(int x) { a = x; } void setB(double y) { b = y; } void setC(double z) { c = z; } public double mul() { return b*c; } public double fn_area() { double result = 0; if(a==1) { result = b*c; }else if(a ==2 ) { result = 0.5*b*c; } return result; } } }
